| 方案 | 优点 | 缺点 |
|------|------|------|
| **Vue3+FastAPI（选这个）** | Alex熟悉Python，团队一致 | 需要自己运维 |

| 微信小程序原生 | 体验好，可分享到朋友圈 | 需要两个代码库，审核周期长 |

### 3.2 数据表详解

#### users（用户表）

```sql
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    nickname        VARCHAR(50) NOT NULL,          -- 微信昵称
    avatar_url     VARCHAR(500),                   -- 微信头像URL
    bio            VARCHAR(200),                   -- 个人介绍
    city           VARCHAR(50),                    -- 所在城市
    industry       VARCHAR(50),                    -- 所在行业
    opc_years      INTEGER DEFAULT 0,             -- OPC经营年数
    credit_score   INTEGER DEFAULT 0,             -- 信誉分
    is_verified    BOOLEAN DEFAULT FALSE,         -- 是否OPC认证
    created_at     TIMESTAMP DEFAULT NOW(),
    updated_at     TIMESTAMP DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_users_credit ON users(credit_score DESC);
CREATE INDEX idx_users_verified ON users(is_verified) WHERE is_verified = TRUE;
```

#### user_auth（认证表）

```sql
CREATE TABLE user_auth (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID UNIQUE REFERENCES users(id) ON DELETE CASCADE,
    auth_type       VARCHAR(20) NOT NULL,         -- 'wechat'
    wechat_openid   VARCHAR(100) UNIQUE NOT NULL, -- 微信OpenID
    wechat_unionid  VARCHAR(100),                  -- 微信UnionID（同一主体下多应用唯一）
    created_at      TIMESTAMP DEFAULT NOW(),
    last_login_at   TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_user_auth_openid ON user_auth(wechat_openid);
```

#### business_verification（OPC认证表）

```sql
CREATE TABLE business_verification (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID UNIQUE REFERENCES users(id) ON DELETE CASCADE,
    company_name    VARCHAR(200) NOT NULL,        -- 公司名称
    unified_credit_code VARCHAR(50) NOT NULL,   -- 统一社会信用代码
    license_front   VARCHAR(500) NOT NULL,       -- 营业执照正面照
    license_back    VARCHAR(500),                  -- 营业执照反面照（可选）
    status          VARCHAR(20) DEFAULT 'pending', -- pending/approved/rejected
    reject_reason   VARCHAR(500),                  -- 拒绝原因
    verified_at     TIMESTAMP,                     -- 认证通过时间
    created_at      TIMESTAMP DEFAULT NOW()
);
```

#### questions（问答表）

```sql
CREATE TABLE questions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE SET NULL,
    title           VARCHAR(200) NOT NULL,         -- 问题标题
    body           TEXT,                           -- 问题详情（Markdown）
    tags            VARCHAR(200)[],                -- 标签数组
    view_count      INTEGER DEFAULT 0,
    answer_count    INTEGER DEFAULT 0,
    is_solved       BOOLEAN DEFAULT FALSE,
    best_answer_id  UUID,                          -- 最佳答案ID
    status          VARCHAR(20) DEFAULT 'active',  -- active/closed/hidden
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_questions_created ON questions(created_at DESC);
CREATE INDEX idx_questions_views ON questions(view_count DESC);
CREATE INDEX idx_questions_solved ON questions(is_solved) WHERE is_solved = TRUE;
-- 全文搜索索引
CREATE INDEX idx_questions_fts ON questions USING gin(to_tsvector('simple', title || ' ' || COALESCE(body, '')));
```

#### answers（回答表）

```sql
CREATE TABLE answers (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    question_id     UUID REFERENCES questions(id) ON DELETE CASCADE,
    user_id         UUID REFERENCES users(id) ON DELETE SET NULL,
    body            TEXT NOT NULL,                  -- 回答内容（Markdown）
    upvote_count    INTEGER DEFAULT 0,
    is_best         BOOLEAN DEFAULT FALSE,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_answers_question ON answers(question_id);
CREATE INDEX idx_answers_upvote ON answers(upvote_count DESC);
```

#### posts（经验分享表）

```sql
CREATE TABLE posts (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE SET NULL,
    post_type       VARCHAR(20) NOT NULL,         -- 'experience'/'resource'/'recruit'/'rant'
    title           VARCHAR(200) NOT NULL,
    body            TEXT,                           -- 内容正文
    images          VARCHAR(500)[],                -- 图片URL数组
    tags            VARCHAR(200)[],                -- 标签
    upvote_count    INTEGER DEFAULT 0,
    comment_count   INTEGER DEFAULT 0,
    favorite_count  INTEGER DEFAULT 0,
    status          VARCHAR(20) DEFAULT 'active',
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_posts_created ON posts(created_at DESC);
CREATE INDEX idx_posts_type ON posts(post_type);
CREATE INDEX idx_posts_upvote ON posts(upvote_count DESC);
CREATE INDEX idx_posts_fts ON posts USING gin(to_tsvector('simple', title || ' ' || COALESCE(body, '')));
```

#### comments（评论表）

```sql
CREATE TABLE comments (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE SET NULL,
    -- 多态关联：可以是回答的评论，也可以是帖子的评论
    target_type     VARCHAR(20) NOT NULL,         -- 'answer'/'post'
    target_id       UUID NOT NULL,
    parent_id       UUID,                          -- 回复某条评论时填这个
    body            TEXT NOT NULL,
    created_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_comments_target ON comments(target_type, target_id);
CREATE INDEX idx_comments_parent ON comments(parent_id);
```

#### likes（点赞表）

```sql
CREATE TABLE likes (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,
    target_type     VARCHAR(20) NOT NULL,         -- 'post'/'answer'
    target_id       UUID NOT NULL,
    created_at      TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, target_type, target_id)
);
```

#### favorites（收藏表）

```sql
CREATE TABLE favorites (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,
    target_type     VARCHAR(20) NOT NULL,         -- 'post'/'question'/'answer'
    target_id       UUID NOT NULL,
    created_at      TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, target_type, target_id)
);
```

#### notifications（通知表）

```sql
CREATE TABLE notifications (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,

    title           VARCHAR(200),
    body            TEXT,

    is_read         BOOLEAN DEFAULT FALSE,
```

## 六、关键实现细节

### 6.1 微信OAuth登录流程

```
用户点击"微信登录"
    ↓
前端调用后端 /api/auth/wechat/login
    ↓
后端返回微信授权URL（含redirect_uri、state参数）
    ↓
前端跳转微信授权页
    ↓
用户授权 → 微信回调 redirect_uri?code=xxx&state=xxx
    ↓
后端用code换OpenID
    ↓
创建/查找user，生成JWT token
    ↓
前端存储token，跳转到首页
```

### 6.2 浏览量计数（防刷）

```python
async def view_question(question_id: str, user_id: str = None):
    """浏览量计数，同一用户对同一问题24小时内只计一次"""
    cache_key = f"view:{question_id}:{user_id}" if user_id else f"view:{question_id}:ip"
    
    if await redis.get(cache_key):
        return  # 已记录，不重复计数
    
    await redis.setex(cache_key, 86400, "1")  # 24小时过期
    await db.execute(
        "UPDATE questions SET view_count = view_count + 1 WHERE id = :id",
        {"id": question_id}
    )
```
